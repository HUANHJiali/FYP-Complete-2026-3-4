# 📋 FYP 操作日志系统设计方案

**设计时间**: 2026-02-07
**设计目标**: 记录用户所有操��行为，提供审计追踪
**设计原则**: 完整性、可追溯性、高性能

---

## 🎯 设计目标

### 核心功能
1. ✅ 记录所有用户登录行为（成功/失败）
2. ✅ 记录关键操作（创建/修改/删除）
3. ✅ 提供日志查询和筛选功能
4. ✅ 支持日志导出
5. ✅ 自动记录，无需手动触发

### 审计需求
- 谁（Who）：用户身份
- 何时（When）：操作时间
- 做了什么（What）：操作类型
- 结果如何（Result）：成功/失败
- 从哪里（Where）：IP地址、设备信息

    operation_detail TEXT COMMENT '操作详情（JSON格式）',
    status TINYINT COMMENT '操作状态：1成功 0失败',
    error_message VARCHAR(500) COMMENT '错误信息（失败时记录）',
    ip_address VARCHAR(50) COMMENT 'IP地址',
    user_agent VARCHAR(500) COMMENT '用户代理（浏览器信息）',
    device_type VARCHAR(50) COMMENT '设备类型：PC/Mobile/Tablet',
    browser_type VARCHAR(50) COMMENT '浏览器类型：Chrome/Firefox/Safari等',
    os_type VARCHAR(50) COMMENT '操作系统：Windows/Mac/Linux/Android/iOS',
    location VARCHAR(100) COMMENT '地理位置（可选）',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '操作时间',
    INDEX idx_user_id (user_id),
    INDEX idx_user_type (user_type),
    INDEX idx_operation_type (operation_type),
    INDEX idx_module_name (module_name),

    INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户操作日志表';

    created_at = models.DateTimeField(auto_now_add=True, verbose_name='操作时间', db_index=True)

### 2. 创建日志工具类

**文件**: `source/server/utils/OperationLogger.py`

```python
import logging
import json
from django.conf import settings
from django.utils import timezone
from models.OperationLog import OperationLog

logger = logging.getLogger(__name__)

class OperationLogger:
    """操作日志记录器"""

    @staticmethod
    def log(request, operation_type, module_name, resource_id=None,
             resource_name=None, status=1, error_message=None, detail=None):
        """
        记录操作日志

        Args:
            request: Django request 对象
            operation_type: 操作类型 (login/logout/create/update/delete/submit等)
            module_name: 模块名称 (exam/question/user等)
            resource_id: 资源ID
            resource_name: 资源名称
            status: 状态 (1:成功 0:失败)
            error_message: 错误信息
            detail: 详细信息字典
        """
        try:
            # 获取用户信息
            user = getattr(request, 'user', None)
            user_id = getattr(user, 'id', 'anonymous')
            user_name = getattr(user, 'name', '匿名用户')
            user_type = getattr(user, 'type', 0)

            # 获取请求信息
            ip_address = OperationLogger._get_client_ip(request)
            user_agent = request.META.get('HTTP_USER_AGENT', '')
            device_info = OperationLogger._parse_user_agent(user_agent)

            # 构建日志记录
            log = OperationLog.objects.create(
                user_id=str(user_id),
                user_name=user_name,
                user_type=user_type,
                operation_type=operation_type,
                module_name=module_name,
                resource_id=str(resource_id) if resource_id else None,
                resource_name=resource_name,
                operation_detail=json.dumps(detail) if detail else None,
                status=status,
                error_message=error_message,
                ip_address=ip_address,
                user_agent=user_agent,
                device_type=device_info.get('device'),
                browser_type=device_info.get('browser'),
                os_type=device_info.get('os'),
            )

            logger.info(f"操作日志已记录: {user_name} - {operation_type}")
            return log

        except Exception as e:
            logger.error(f"记录操作日志失败: {e}")
            return None

    @staticmethod
    def log_login(request, status, error_message=None):
        """记录登录日志"""
        user_id = getattr(request, 'user', None)
        user_name = request.POST.get('userName', 'unknown')
        detail = {
            'login_method': 'password',
            'login_time': timezone.now().isoformat()
        }
        return OperationLogger.log(
            request=request,
            operation_type='login',
            module_name='auth',
            status=status,
            error_message=error_message,
            detail=detail
        )

    @staticmethod
    def log_logout(request):
        """记录登出日志"""
        return OperationLogger.log(
            request=request,
            operation_type='logout',
            module_name='auth',
            status=1
        )

    @staticmethod
    def log_create(request, module_name, resource_id, resource_name, detail=None):
        """记录创建操作"""
        return OperationLogger.log(
            request=request,
            operation_type='create',
            module_name=module_name,
            resource_id=resource_id,
            resource_name=resource_name,
            detail=detail
        )

    @staticmethod
    def log_update(request, module_name, resource_id, resource_name, detail=None):
        """记录更新操作"""
        return OperationLogger.log(
            request=request,
            operation_type='update',
            module_name=module_name,
            resource_id=resource_id,
            resource_name=resource_name,
            detail=detail
        )

    @staticmethod
    def log_delete(request, module_name, resource_id, resource_name, detail=None):
        """记录删除操作"""
        return OperationLogger.log(
            request=request,
            operation_type='delete',
            module_name=module_name,
            resource_id=resource_id,
            resource_name=resource_name,
            detail=detail
        )

    @staticmethod
    def _get_client_ip(request):
        """获取客户端IP地址"""
        x_forwarded_for = request.META.get('HTTP_X_FORWARDED_FOR')
        if x_forwarded_for:
            ip = x_forwarded_for.split(',')[0]
        else:
            ip = request.META.get('REMOTE_ADDR')
        return ip

    @staticmethod
    def _parse_user_agent(user_agent):
        """解析User-Agent，提取设备信息"""
        device_type = 'PC'
        browser_type = 'Unknown'
        os_type = 'Unknown'

        user_agent_lower = user_agent.lower()

        # 检测设备类型
        if 'mobile' in user_agent_lower or 'android' in user_agent_lower:
            device_type = 'Mobile'
        elif 'tablet' in user_agent_lower or 'ipad' in user_agent_lower:
            device_type = 'Tablet'

        # 检测浏览器
        if 'chrome' in user_agent_lower:
            browser_type = 'Chrome'
        elif 'firefox' in user_agent_lower:
            browser_type = 'Firefox'
        elif 'safari' in user_agent_lower and 'chrome' not in user_agent_lower:
            browser_type = 'Safari'
        elif 'edge' in user_agent_lower:
            browser_type = 'Edge'
        elif 'msie' in user_agent_lower or 'trident' in user_agent_lower:
            browser_type = 'IE'

        # 检测操作系统
        if 'windows' in user_agent_lower:
            os_type = 'Windows'
        elif 'mac' in user_agent_lower and 'iphone' not in user_agent_lower:
            os_type = 'Mac'
        elif 'linux' in user_agent_lower:
            os_type = 'Linux'
        elif 'android' in user_agent_lower:
            os_type = 'Android'
        elif 'iphone' in user_agent_lower or 'ios' in user_agent_lower:
            os_type = 'iOS'

        return {
            'device': device_type,
            'browser': browser_type,
            'os': os_type
        }
```

### 3. 创建日志查询API

**文件**: `source/server/views/LogViews.py`

```python
from django.db.models import Q
from django.core.paginator import Paginator
from utils.BaseView import BaseView
from models.OperationLog import OperationLog
from utils.OperationLogger import OperationLogger
import json

class OperationLogView(BaseView):
    """操作日志视图"""

    def get(self, request):
        """查询操作日志"""
        try:
            # 获取筛选参数
            page = int(request.GET.get('page', 1))
            limit = int(request.GET.get('limit', 20))
            user_id = request.GET.get('userId')
            user_type = request.GET.get('userType')
            operation_type = request.GET.get('operationType')
            module_name = request.GET.get('moduleName')
            status = request.GET.get('status')
            start_date = request.GET.get('startDate')
            end_date = request.GET.get('endDate')

            # 构建查询条件
            query = Q()

            if user_id:
                query &= Q(user_id__icontains=user_id)
            if user_type:
                query &= Q(user_type=user_type)
            if operation_type:
                query &= Q(operation_type=operation_type)
            if module_name:
                query &= Q(module_name=module_name)
            if status is not None:
                query &= Q(status=status)
            if start_date:
                query &= Q(created_at__gte=start_date)
            if end_date:
                query &= Q(created_at__lte=end_date)

            # 查询日志
            logs = OperationLog.objects.filter(query).order_by('-created_at')

            # 分页
            paginator = Paginator(logs, limit)
            logs_page = paginator.get_page(page)

            # 序列化数据
            log_list = []
            for log in logs_page:
                log_list.append({
                    'id': log.id,
                    'userId': log.user_id,
                    'userName': log.user_name,
                    'userType': log.user_type,
                    'operationType': log.operation_type,
                    'moduleName': log.module_name,
                    'resourceId': log.resource_id,
                    'resourceName': log.resource_name,
                    'operationDetail': log.operation_detail,
                    'status': log.status,
                    'errorMessage': log.error_message,
                    'ipAddress': log.ip_address,
                    'deviceType': log.device_type,
                    'browserType': log.browser_type,
                    'osType': log.os_type,
                    'createdAt': log.created_at.strftime('%Y-%m-%d %H:%M:%S')
                })

            return self.success({
                'list': log_list,
                'total': logs.count(),
                'page': page,
                'limit': limit,
                'totalPages': paginator.num_pages
            })

        except Exception as e:
            return self.error(f'查询日志失败: {str(e)}')

    def delete(self, request):
        """删除日志（批量）"""
        try:
            log_ids = request.data.get('ids', [])

            if not log_ids:
                return self.error('请选择要删除的日志')

            # 删除日志
            count = OperationLog.objects.filter(id__in=log_ids).delete()[0]

            # 记录删除操作
            OperationLogger.log_delete(
                request=request,
                module_name='log',
                resource_id=str(log_ids),
                resource_name=f'{count}条日志',
                detail={'count': count}
            )

            return self.success(f'成功删除 {count} 条日志')

        except Exception as e:
            return self.error(f'删除日志失败: {str(e)}')

    def export(self, request):
        """导出日志"""
        import csv
        from django.http import HttpResponse

        try:
            # 获取筛选参数（与get方法相同）
            query = Q()
            # ... (省略筛选逻辑)

            # 查询日志
            logs = OperationLog.objects.filter(query).order_by('-created_at')[:10000]

            # 创建CSV响应
            response = HttpResponse(content_type='text/csv')
            response['Content-Disposition'] = 'attachment; filename="operation_logs.csv"'

            writer = csv.writer(response)

            # 写入表头
            writer.writerow([
                '日志ID', '用户ID', '用户姓名', '用户类型', '操作类型',
                '模块名称', '资源ID', '资源名称', '状态', '错误信息',
                'IP地址', '设备类型', '浏览器', '操作系统', '操作时间'
            ])

            # 写入数据
            for log in logs:
                writer.writerow([
                    log.id,
                    log.user_id,
                    log.user_name,
                    log.get_user_type_display(),
                    log.operation_type,
                    log.module_name,
                    log.resource_id or '',
                    log.resource_name or '',
                    '成功' if log.status == 1 else '失败',
                    log.error_message or '',
                    log.ip_address or '',
                    log.device_type or '',
                    log.browser_type or '',
                    log.os_type or '',
                    log.created_at.strftime('%Y-%m-%d %H:%M:%S')
                ])

            return response

        except Exception as e:
            return self.error(f'导出日志失败: {str(e)}')
```

export default {
  name: 'AdminLogs',
  data() {
    return {
      loading: false,
      logList: [],
      selectedLogs: [],
      showDetailModal: false,
      currentLog: null,
      filters: {
        userId: '',
        userType: '',
        operationType: '',
        moduleName: '',
        status: '',
        dateRange: []
      },
      pagination: {
        current: 1,
        pageSize: 20,
        total: 0
      },
      columns: [
        { type: 'selection', width: 60, align: 'center' },
        { title: '日志ID', key: 'id', width: 80 },
        { title: '用户姓名', key: 'userName', width: 100 },
        { title: '用户类型', key: 'userType', width: 90, slot: 'userType' },
        { title: '操作类型', key: 'operationType', width: 90, slot: 'operationType' },
        { title: '模块', key: 'moduleName', width: 90 },
        { title: '资源名称', key: 'resourceName', ellipsis: true },
        { title: '状态', key: 'status', width: 70, slot: 'status' },
        { title: 'IP地址', key: 'ipAddress', width: 120 },
        { title: '设备', key: 'deviceType', width: 80 },
        { title: '操作时间', key: 'createdAt', width: 160 },
        { title: '操作', key: 'action', width: 80, slot: 'action', align: 'center' }
      ]
    }
  },
  methods: {
    async searchLogs() {
      this.loading = true
      try {
        const params = {
          page: this.pagination.current,
          limit: this.pagination.pageSize,
          ...this.filters
        }

        const response = await getOperationLogs(params)

        if (response.code === 0) {
          this.logList = response.data.list
          this.pagination.total = response.data.total
        } else {
          this.$Message.error(response.msg)
        }
      } catch (error) {
        this.$Message.error('查询日志失败')
      } finally {
        this.loading = false
      }
    },

    resetFilters() {
      this.filters = {
        userId: '',
        userType: '',
        operationType: '',
        moduleName: '',
        status: '',
        dateRange: []
      }
      this.pagination.current = 1
      this.searchLogs()
    },

    viewDetail(log) {
      this.currentLog = log
      this.showDetailModal = true
    },

    onSelectionChange(selection) {
      this.selectedLogs = selection
    },

    batchDelete() {
      if (this.selectedLogs.length === 0) {
        this.$Message.warning('请选择要删除的日志')
        return
      }

      this.$Modal.confirm({
        title: '确认删除',
        content: `确定要删除选中的 ${this.selectedLogs.length} 条日志吗？`,
        onOk: async () => {
          try {
            const ids = this.selectedLogs.map(log => log.id)
            const response = await deleteOperationLogs({ ids })

            if (response.code === 0) {
              this.$Message.success('删除成功')
              this.searchLogs()
            } else {
              this.$Message.error(response.msg)
            }
          } catch (error) {
            this.$Message.error('删除失败')
          }
        }
      })
    },

    async exportLogs() {
      try {
        const params = { ...this.filters }
        window.open(`/api/logs/export/?${new URLSearchParams(params)}`)
      } catch (error) {
        this.$Message.error('导出失败')
      }
    },

    getOperationTypeLabel(type) {
      const labels = {
        login: '登录',
        logout: '登出',
        create: '创建',
        update: '更新',
        delete: '删除',
        submit: '提交'
      }
      return labels[type] || type
    },

    getOperationTypeColor(type) {
      const colors = {
        login: 'success',
        logout: 'default',
        create: 'primary',
        update: 'warning',
        delete: 'error',
        submit: 'info'
      }
      return colors[type] || 'default'
    },

    getUserTypeLabel(type) {
      const labels = { 1: '管理员', 2: '教师', 3: '学生' }
      return labels[type] || '未知'
    },

    getUserTypeColor(type) {
      const colors = { 1: 'error', 2: 'warning', 3: 'primary' }
      return colors[type] || 'default'
    }
  },
  mounted() {
    this.searchLogs()
  }
}
</script>
